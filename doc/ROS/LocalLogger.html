def initialize(io)
  @logger = Logger.new(io)
  # add color escape sequence
  @logger.formatter = proc {|severity, datetime, progname, msg|
    header = ''
    footer = ''
    if severity == 'ERROR' or severity == 'FATAL' 
      header = "\e[31m"
      footer = "\e[0m"
    elsif severity == 'WARN'
      header = "\e[33m"
      footer = "\e[0m"
    end
    header + "[#{severity}] #{msg}" + footer + "\n"
  }
end
