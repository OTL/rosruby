def <=>(other)
  diff = self.to_nsec - other.to_nsec
  if diff > 0
    1
  elsif diff < 0
    -1
  else
    0
  end
end

def canonicalize
  while @nsecs >= 1e9.to_i
    @secs += 1
    @nsecs -= 1e9.to_i
  end
  while @nsecs < 0
    @secs -= 1
    @nsecs += 1e9.to_i
  end
  self
end
