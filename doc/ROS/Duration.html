def initialize(secs=0, nsecs=nil)
  @secs = secs.to_i
  if nsecs
    @nsecs = nsecs
  else
    @nsecs = ((secs - @secs) * 1e9.to_i).to_i
  end
  canonicalize
end

def +(duration)
  tm = ::ROS::Duration.new
  tm.secs = @secs + duration.secs
  tm.nsecs = @nsecs + duration.nsecs
  tm.canonicalize
end
