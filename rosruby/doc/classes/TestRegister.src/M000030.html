  def test_publisher
    proxy = ROS::MasterProxy.new('/test_node3', ENV['ROS_MASTER_URI'], 'http://dummy:11111')
    sub = proxy.register_publisher('/rosout', 'rosgraph_msgs/Log')
    assert(sub.length > 0)
    num = proxy.unregister_publisher('/rosout')
    assert_equal(1, num)
  end
