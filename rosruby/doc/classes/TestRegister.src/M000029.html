  def test_subscriber
    proxy = ROS::MasterProxy.new('/test_node2', ENV['ROS_MASTER_URI'], 'http://dummy:11111')
    pub = proxy.register_subscriber('/rosout_agg', 'rosgraph_msgs/Log')
    assert(pub.length > 0)
    proxy.unregister_subscriber('/rosout_agg')
  end
