  def test_anonymous
    node1 = ROS::Node.new('/test_anonymous1', true)
    assert_not_equal('/test_anonymous1', node1.node_name)
    node2 = ROS::Node.new('/test_anonymous1', true)
    assert_not_equal('/test_anonymous1', node2.node_name)
    sleep(0.5)
    assert(node1.ok?)
    assert(node2.ok?)
    node1.shutdown
    node2.shutdown
    assert(!node1.ok?)
    assert(!node2.ok?)
  end
