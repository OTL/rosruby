  def test_param_subscribe
    node = ROS::Node.new('/test_param_sub')
    called = false
    node.subscribe_parameter('/test_param1') do |param|
      called = param
    end
    node.set_param('/test_param1', 1)
    sleep(0.1)
    assert_equal(1, called)
    node.shutdown
  end
