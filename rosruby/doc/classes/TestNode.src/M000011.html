  def test_up_down
    node1 = ROS::Node.new('/test_up_down')
    assert_equal('/test_up_down', node1.node_name)
    assert(node1.ok?)
    node1.shutdown
    assert(!node1.ok?)
    node2 = ROS::Node.new('/test2')
    assert(node2.ok?)
    node2.shutdown
    assert(!node2.ok?)
  end
