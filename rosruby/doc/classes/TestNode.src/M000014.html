  def test_not_anonymous
    node1 = ROS::Node.new('/test_not_anonymous1')
    node2 = ROS::Node.new('/test_not_anonymous1')
    sleep(0.5)
    assert(!node1.ok?) # killed by master
    assert(node2.ok?)
    node1.shutdown
    node2.shutdown
    assert(!node2.ok?)
  end
