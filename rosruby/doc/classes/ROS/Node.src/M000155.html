    def advertise_service(service_name, service_type, &callback)
      server = ::ROS::ServiceServer.new(@node_name,
                                        resolve_name(service_name),
                                        service_type,
                                        callback)
      @manager.add_service_server(server)
      trap_signals
      server
    end
