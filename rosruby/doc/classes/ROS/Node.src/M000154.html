    def advertise(topic_name, topic_type, latched=false, resolve=true)
      if resolve
        name = resolve_name(topic_name)
      else
        name = topic_name
      end
      publisher = Publisher.new(@node_name,
                                name,
                                topic_type,
                                latched,
                                @manager.host)
      @manager.add_publisher(publisher)
      trap_signals
      publisher
    end
