    def build_header
      header = Header.new
      header.push_data("callerid", @caller_id)
      header.push_data("service", @service_name)
      header.push_data("md5sum", @service_type.md5sum)
      header.push_data("type", @service_type.type)
      if @persistent
        header.push_data("persistent", '1')
      end
      header
    end
