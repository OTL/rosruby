    def build_header
      header = Header.new
      header["callerid"] = @caller_id
      header["topic"] = @topic_name
      header["md5sum"] = @topic_type.md5sum
      header["type"] = @topic_type.type
      if latching?
        header["latching"]  = '1'
      end
      header
    end
