    def deserialize(data)
      while data.length > 0
        len, data = data.unpack('Va*')
        msg = data[0..(len-1)]
        equal_position = msg.index('=')
        key = msg[0..(equal_position-1)]
        value = msg[(equal_position+1)..-1]
        @data[key] = value
        data = data[(len)..-1]
      end
      self
    end
