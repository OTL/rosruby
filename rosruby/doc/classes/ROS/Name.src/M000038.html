    def anonymous_name(id)
      # Generate a ROS-legal 'anonymous' name
      # @param id: prefix for anonymous name
      # @type  id: str
      name = "#{id}_#{Socket.gethostname}_#{Process.pid}_#{rand(1000000)}"
      name = name.gsub('.', '_')
      name = name.gsub('-', '_')
      name.gsub(':', '_')
    end
