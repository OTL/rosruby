    def add_path_of_package(package)
      path = @@all_packages[package]
      ["#{path}/msg_gen/ruby", "#{path}/srv_gen/ruby", "#{path}/lib"].each do |path|
        if File.exists?(path)
          if not $:.include?(path)
            $:.push(path)
          end
        end
      end
    end
